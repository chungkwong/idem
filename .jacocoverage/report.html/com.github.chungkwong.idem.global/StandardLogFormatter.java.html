package com.github.chungkwong.idem.global;
import java.text.*;
import java.util.logging.*;
/**
 *
 * @author kwong
 */
public class StandardLogFormatter extends Formatter{
	static final DateFormat datefmt=DateFormat.getDateTimeInstance(DateFormat.SHORT,DateFormat.SHORT);
	@Override
	public String format(LogRecord record){
		StringBuilder buf=new StringBuilder();
		buf.append('[').append(record.getLevel().getName()).append(']');
		buf.append(record.getThreadID()).append(':');
		buf.append(record.getSourceClassName()).append('.');
		buf.append(record.getSourceMethodName()).append(':');
		buf.append(record.getMessage()).append('\n');
		if(record.getParameters()!=null){
			int i=0;
			for(Object obj:record.getParameters())
				buf.append('$').append(i).append('=').append(obj).append('\n');
		}
		if(record.getThrown()!=null)
			buf.append(Log.throwableToString(record.getThrown()));
		return buf.toString();
	}
}
